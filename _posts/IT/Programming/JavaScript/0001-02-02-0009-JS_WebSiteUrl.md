> window.location를 이용하여 현재 접속 중인 웹사이트의 주소(URL)을 확인한다.

### 웹페이지 주소 정보 확인
 
```javascript
    /**
        예) https://www.google.com/search?keyword=3
     */

    // 1) 현재 페이지의 URL 정보
        let hrefInfo = window.location.href;
        console.log(hrefInfo);          // 'https://www.google.com/search?keyword=3'
        
    // 2) 웹 호스트의 도메인 정보
        let hostnameInfo = window.location.hostname;
        console.log(hostnameInfo);      // 'www.google.com'
        
        let hostInfo = window.location.host;
        console.log(hostInfo);          // 'www.google.com' hostname과 같지만  ':', 포트 번호를 포함한다.
        
    // 3) 현재 페이지의 경로 정보, '/' 뒤 URL의 경로
        let pathnameInfo = window.location.pathname;
        console.log(pathnameInfo);      // '/search'
        
    // 4) 프로토콜 정보 (http: or https:)
        let protocolInfo = window.location.protocol;
        console.log(protocolInfo);      // 'https:'
        
    // 5) QueryString의 파라미터 정보. '?' 문자 뒤 URL의 QueryString
        let searchInfo = window.location.search;
        console.log(searchInfo);        // '?keyword=3'
    // 모던 브라우저에서는 URLSearchParams.get()과 URL.searchParams를 사용해서 인자를 쉽게 추출 할 수 있다.
        
    // 6) 표준 형태
        let originInfo = window.location.origin;
        console.log(originInfo);        // 'https://www.google.com'
        
    // 7) '#' 문자 뒤 URL
        let hashInfo = window.location.hash;
        console.log(hashInfo);          // 
        
```
